CURSOR CUR_BIIS_PERSON
IS
SELECT HHSID,
			LAST_NAME,
			FIRST_NAME,
			MIDDLE_NAME,
			LOWER(EMAIL_ADDR) AS EMAIL_ADDR,
			USERTYPE,
			PIV_CARD_STATUS,
			SCMS_STATUS,
			EMPLID,
			PAY_PLAN,
			GRADE,
			POSITION_TITLE_NAME,
			SERIES,
			BARG_UNIT,
			OPDIV,
			ORG_CD,
			PAY_BASIS,
			STEP,
			BASIC_PAY,
			ADJ_BASIC_PAY,
			CAN_CD,
			JOB_CODE,
			POSITION_NUM,
			EMP_STATUS_FROM_PERS_SYSTEM,
			LAST_UPDT_FROM_PERS_SYSTEM_DT
	FROM HISTDBA.EWITS_BIIS_PERSON_MV@BIIS_DBLINK
	WHERE OPDIV IN ('CDC', 'IHS', 'OS', 'ACF', 'ACL','AHRQ', 'PSC', 'SAMHSA')
		AND HHSID IS NOT NULL
		AND EMAIL_ADDR IS NOT NULL
		AND ((USERTYPE = 'Employee' AND EMP_STATUS_FROM_PERS_SYSTEM != 'Terminated')
		OR (USERTYPE != 'Employee' AND EMP_STATUS_FROM_PERS_SYSTEM = 'Active'))
    UNION
    SELECT HHSID,
			LAST_NAME,
			FIRST_NAME,
			MIDDLE_NAME,
			LOWER(EMAIL_ADDR) AS EMAIL_ADDR,
			USERTYPE,
			PIV_CARD_STATUS,
			SCMS_STATUS,
			EMPLID,
			PAY_PLAN,
			GRADE,
			POSITION_TITLE_NAME,
			SERIES,
			BARG_UNIT,
			OPDIV,
			ORG_CD,
			PAY_BASIS,
			STEP,
			BASIC_PAY,
			ADJ_BASIC_PAY,
			CAN_CD,
			JOB_CODE,
			POSITION_NUM,
			EMP_STATUS_FROM_PERS_SYSTEM,
			LAST_UPDT_FROM_PERS_SYSTEM_DT
	FROM HISTDBA.EWITS_BIIS_PERSON_MV@BIIS_DBLINK
	WHERE OPDIV ='CMS'
		AND HHSID IS NOT NULL
		AND EMAIL_ADDR IS NOT NULL
		AND ((USERTYPE = 'Employee' AND EMP_STATUS_FROM_PERS_SYSTEM IN('Terminated','Active')
        AND LAST_UPDT_FROM_PERS_SYSTEM_DT >= add_months(SYSDATE, -3))
		OR (USERTYPE != 'Employee' AND EMP_STATUS_FROM_PERS_SYSTEM = 'Active'));
